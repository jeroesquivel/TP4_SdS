% -- Sistema particular -------------------------------------------------------
\begin{frame}{Sistema particular}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \begin{itemize}\setlength\itemsep{2pt}
                \item Recinto circular de diámetro $L = \SI{80}{m}$.
                \item Obstáculo fijo central, radio $r_0 = \SI{1}{m}$.
                \item $N$ partículas; $r = \SI{1}{m}$, $m = \SI{1}{kg}$.
                \item $|\vect{v}_0| = \SI{1}{m/s}$, dirección uniforme $\in [0,2\pi)$.
                \item Distribución inicial uniforme, sin solapamiento.
                \item Fuerza elástica blanda de a pares:
                \begin{itemize}
                    \item $\vect{F}_{ij} = -k\,\xi\,\hat{\vect{e}}_{ij}$ si $\xi > 0$.
                    \item $\xi = r_i + r_j - |\vect{x}_i - \vect{x}_j|$.
                \end{itemize}
                \item Obstáculo y borde: masa infinita, misma forma.
                \item Estados:
                \begin{itemize}
                    \item \textcolor{green!55!black}{\textbf{Fresca}}: toca obstáculo $\rightarrow$ usada.
                    \item \textcolor{violet!70!black}{\textbf{Usada}}: toca borde $\rightarrow$ fresca.
                \end{itemize}
            \end{itemize}
        \end{column}
        \begin{column}{0.45\textwidth}
            \centering
            \begin{tikzpicture}[scale=0.85]
                \draw[thick] (0,0) circle (3);
                \fill[violet!70!black] (0,0) circle (0.25);
                \foreach \x/\y in {1.2/0.5, -1.5/0.8, 0.3/-1.8, 1.8/-1.2, -0.9/-2.1, -2.0/-0.4, 0.7/2.1, 1.9/1.5}
                \fill[green!60!black] (\x,\y) circle (0.12);
                \foreach \x/\y in {-0.5/1.6, 2.2/0.3, -1.8/1.8}
                \fill[violet!70!black] (\x,\y) circle (0.12);
                \draw[<->] (-3,-3.4) -- (3,-3.4) node[midway,below,font=\small]{$L$};
                \node[font=\scriptsize] at (0.5,0.15) {$r_0$};
            \end{tikzpicture}
        \end{column}
    \end{columns}
\end{frame}

% -- Implementación: solo motor -----------------------------------------------
\begin{frame}{Motor de simulación}
    \begin{enumerate}\setlength\itemsep{2pt}
        \item Sembrar $N$ partículas (rejection sampling): posiciones sin solapamiento; $|\vect{v}| = v_0$, ángulo uniforme.
        \item Indexar en \emph{Cell Index Method} (celda $= 2r$): vecinos en $\mathcal{O}(1)$.
        \item \textbf{Mientras} $t < t_f$:
        \begin{enumerate}\setlength\itemsep{1pt}
            \item Calcular fuerzas: pares en contacto ($\xi>0$), obstáculo, borde. \hfill $\mathcal{O}(N)$
            \item Integrar Velocity-Verlet 2D un paso $\Delta t$. \hfill $\mathcal{O}(N)$
            \item Actualizar flags de contacto y estado (fresca/usada).
            \item Muestrear cada $\Delta t_2$ para perfiles; $C_{fc}$ con resolución $\Delta t$.
        \end{enumerate}
    \end{enumerate}
    \vspace{0.3cm}
    \textbf{Integrador:} Velocity-Verlet 2D (symplectic, $\vect{F}(\vect{x})$ sin dependencia en $\vect{v}$).
    \framecite{Teórica 4, slide 17}
\end{frame}

% -- Observables --------------------------------------------------------------
\begin{frame}{Observables}
    \textbf{Cambio de estado:}
    \[
        C_{fc}(t) = \#\{\text{transiciones fresca}\to\text{usada en }[0,t]\},
        \qquad
        J = \left.\frac{\mathrm{d}\,C_{fc}}{\mathrm{d}t}\right|_{\text{estac.}}
    \]
    \vspace{-0.1cm}
    Sólo se cuenta el \textbf{primer $\Delta t$} de cada contacto con el obstáculo.

    \vspace{0.3cm}
    \textbf{Perfiles radiales} (capas $\mathrm{d}S = \SI{0.2}{m}$; partículas frescas con $\vect{x}_j\!\cdot\!\vect{v}_j < 0$):
    \[
        \langle\rho_f^{\,\text{in}}\rangle(S) = \frac{1}{M\,T}\sum_{k,t}\frac{n_k^{\,\text{in}}(S,t)}{A(S)},
        \quad
        |\langle v_f^{\,\text{in}}\rangle(S)| = \left|\frac{1}{M\,T}\sum_{k,t}\frac{1}{n_k^{\,\text{in}}}\!\!\sum_{j\in P_f^{\,\text{in}}}\!\!\tfrac{\vect{x}_j\!\cdot\!\vect{v}_j}{|\vect{x}_j|}\right|
    \]
    \[
        J_{\text{in}}(S) = \langle\rho_f^{\,\text{in}}\rangle(S)\,|\langle v_f^{\,\text{in}}\rangle(S)|
    \]
\end{frame}

\section{Conclusiones}

\begin{frame}{Conclusiones}
    \begin{itemize}\setlength\itemsep{4pt}
        \item \textbf{Sistema 1}: Gear-5 con $\alpha_0 = 3/16$ minimiza el ECM en todo el rango de $\Delta t$ estudiado; pendientes en log-log consistentes con el orden teórico de cada integrador.
        \item \textbf{Sistema 2}: el scanning rate $\langle J\rangle$ crece con $N$ y satura; el régimen cerca del obstáculo está dominado por la densidad de partículas frescas entrantes.
        \item La fuerza elástica blanda con $k$ creciente reproduce el límite event-driven del TP3.
        \item DM time-driven escala mejor con $N$ que EDMD en el rango estudiado.
    \end{itemize}
\end{frame}
